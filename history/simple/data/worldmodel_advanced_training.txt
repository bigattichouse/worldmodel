def is_prime(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True

num = 97
result = is_prime(num)
print(f"{num} is {'prime' if result else 'not prime'}")
</model>
