def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

User: Implement binary search to find 90 in sorted array [11, 11, 14, 16, 21, 27, 44, 53, 77, 90]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [11, 11, 14, 16, 21, 27, 44, 53, 77, 90]

result = binary_search(arr, target)
if result != -1:

User: Implement binary search to find 48 in sorted array [22, 42, 42, 47, 48, 63, 65, 71, 76, 86]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [22, 42, 42, 47, 48, 63, 65, 71, 76, 86]
target = 48
result = binary_search(arr, target)
if result != -1:
    print(f"Found {target} at index {result}")
else:
    print(f"{target} not found in array")
</model>
<requires>python:algorithms</requires>

Binary search for 48 in the array.

User: Implement binary search to find 27 in sorted array [11, 25, 27, 59, 60, 66, 88, 90, 97, 100]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

User: Implement binary search to find 64 in sorted array [5, 10, 23, 24, 27, 29, 38, 51, 58, 64]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:

User: Implement binary search to find 64 in sorted array [21, 30, 31, 52, 63, 64, 65, 66, 79, 95]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [21, 30, 31, 52, 63, 64, 65, 66, 79, 95]
target = 64
result = binary_search(arr, target)
if result != -1:
    print(f"Found {target} at index {result}")

User: Implement binary search to find 30 in sorted array [2, 13, 30, 35, 40, 44, 61, 66, 72, 98]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [2, 13, 30, 35, 40, 44, 61, 66, 72, 98]
target = 30
result = binary_search(arr, target)
if result != -1:
    print(f"Found {target} at index {result}")
else:
    print(f"{target} not found in array")

User: Implement binary search to find 78 in sorted array [6, 8, 23, 24, 38, 51, 60, 68, 92, 94]
Assistant: <think>Binary search repeatedly divides the search space in half. Compare target with middle element and eliminate half the array.</think>
<model>
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

arr = [6, 8, 23, 24, 38, 51, 60, 68, 92, 94]
target = 78
result = binary_search(arr, target)
if result != -1:
    print(f"Found {target} at index {result}")
else:
    print(f"{target} not found in array")
</model>
